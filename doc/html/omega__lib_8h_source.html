32  double get_J( const dealii::Vector<double>& F,
33  const bool symmetric = false)
34  {
35  if(!symmetric)
36  {
37  return F[0] * F[4] * F[8]
38  + F[1] * F[5] * F[6]
39  + F[2] * F[3] * F[7]
40  - F[2] * F[4] * F[6]
41  - F[1] * F[3] * F[8]
42  - F[0] * F[5] * F[7];
43  }
44  else
45  {
46  return F[0] * F[3] * F[5]
47  + F[1] * F[2] * F[4] * 2.0
48  - F[0]*F[4]*F[4]
49  - F[1]*F[1]*F[5]
50  - F[2]*F[2]*F[3];
51  }
52  }
