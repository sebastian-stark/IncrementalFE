36  double get_J( const dealii::Vector<double>& F,
37  const bool symmetric = false)
38  {
39  if(!symmetric)
40  {
41  return F[0] * F[4] * F[8]
42  + F[1] * F[5] * F[6]
43  + F[2] * F[3] * F[7]
44  - F[2] * F[4] * F[6]
45  - F[1] * F[3] * F[8]
46  - F[0] * F[5] * F[7];
47  }
48  else

50  return F[0] * F[3] * F[5]
51  + F[1] * F[2] * F[4] * 2.0
52  - F[0]*F[4]*F[4]
53  - F[1]*F[1]*F[5]
54  - F[2]*F[2]*F[3];
