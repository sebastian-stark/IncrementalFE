35  double get_J( const dealii::Vector<double>& F,
36  const bool symmetric = false)
37  {
38  if(!symmetric)
39  {
40  return F[0] * F[4] * F[8]
41  + F[1] * F[5] * F[6]
42  + F[2] * F[3] * F[7]
43  - F[2] * F[4] * F[6]
44  - F[1] * F[3] * F[8]
