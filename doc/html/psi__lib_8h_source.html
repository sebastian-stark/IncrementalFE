37  double get_J( const dealii::Vector<double>& F,
38  const bool symmetric = false)
39  {
40  if(!symmetric)
41  {
42  return F[0] * F[4] * F[8]
43  + F[1] * F[5] * F[6]
44  + F[2] * F[3] * F[7]
45  - F[2] * F[4] * F[6]
46  - F[1] * F[3] * F[8]
47  - F[0] * F[5] * F[7];
48  }
49  else
50  {
51  return F[0] * F[3] * F[5]
52  + F[1] * F[2] * F[4] * 2.0
53  - F[0]*F[4]*F[4]
54  - F[1]*F[1]*F[5]
55  - F[2]*F[2]*F[3];
56  }
