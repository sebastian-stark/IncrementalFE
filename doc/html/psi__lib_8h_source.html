587  const std::tuple<bool, bool, bool> requested_quantities)
588  const
589  {
590 
591  static dealii::Tensor<2, 3> F, P, T, E, I, B;
592  for(unsigned int i = 0; i < 3; ++i)
593  for(unsigned int j = 0; j < 3; ++j)
594  F[i][j] = values[i*3 + j];
595  const double c = values[9];
596 
597  I = dealii::unit_symmetric_tensor<3,double>();
598  E = 0.5 * (dealii::contract<0, 0>(F, F) - I) - deps/3.0 * (c/c_ref - 1.0) * I;
599  T = lambda * dealii::trace(E) * I + 2.0 * mu * E;
600 
601  if(get<0>(requested_quantities))
602  {
603  omega = 0.5 * lambda * dealii::trace(E) * dealii::trace(E) + mu * trace(dealii::contract<0, 0>(E, E));
604  }
605 
606  if(get<1>(requested_quantities))
607  {
608 
609  P = dealii::contract<1,0>(F, T);
610  for(unsigned int i = 0; i < 3; ++i)
611  for(unsigned int j = 0; j < 3; ++j)
612  d_omega[i*3 + j] = P[i][j];
613  d_omega[9] = -deps/c_ref * trace(E) * (lambda + 2.0 * mu / 3.0);
614  }
615 
616  if(get<2>(requested_quantities))
617  {
618  B = dealii::contract<1, 1>(F, F);
619  for(unsigned int i = 0; i < 3; ++i)
620  for(unsigned int j = 0; j < 3; ++j)
621  {
622  for(unsigned int k = 0; k < 3; ++k)
623  for(unsigned int l = 0; l < 3; ++l)
624  d2_omega[i*3 + j][k*3 + l] = I[i][k] * T[l][j] + lambda * F[i][j] * F[k][l] + mu * ( F[i][l] * F[k][j] + I[j][l] * B[k][i] );
625  d2_omega[i*3 + j][9] = d2_omega[9][i*3 + j] = -deps/c_ref * (lambda + 2.0 * mu / 3.0) * F[i][j];
626  }

628  }
